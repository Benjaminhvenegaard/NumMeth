"REAL""PROCEDURE"CHEPOL(N,X);
"VALUE"N,X;"INTEGER"N;"REAL"X;

   T2:=X;T1:=1;X2:=X+X;
   "FOR"I:=2"STEP"1"UNTIL"N"DO"
   "BEGIN"H:=X2*T2-T1;T1:=T2;T2:=H"END";
   CHEPOL:=H
